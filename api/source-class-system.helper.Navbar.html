39     public function buildMenu($items, $level = 0) {
40         $ret = "";
41         $indent = str_repeat(" ", $level * 2);
42         $ul = ($level > 0) ? $this->styleUlSubMenu : $this->styleUl;
43         $ret .= "$indent<ul {$ul}>\n";
44         $indent = str_repeat(" ", ++$level * 2);
45         foreach ($items as $item => $subitems) {
46             $title = $subitems['title'];
47             $url = $subitems['url'];
48             $visible = (isset($subitems['visible'])) ? $subitems['visible'] : TRUE;
49             $submenu = (isset($subitems['submenu'])) ? $subitems['submenu'] : '';
50             if (!$visible)
51                 continue;
52             $li = (isset($subitems['submenu'])) ? $this->styleLlSubMenu : $this->styleLl;
53             $link = (isset($subitems['submenu'])) ? $this->styleLinkSubmenu : $this->styleLink;
54             $ret .= "$indent<li $li><a href='$url' $link>$title</a>";
55 
56             if (is_array($submenu)) {
57                 $ret .= "\n";
58                 $ret .= $this->buildMenu($submenu, $level + 1);
59                 $ret .= $indent;
60             }
61             $ret .= sprintf("</li>\n", $indent);
62         }
63         $indent = str_repeat(" ", --$level * 2);
64         $ret .= sprintf("%s</ul>\n", $indent);
65         return($ret);
66     }
