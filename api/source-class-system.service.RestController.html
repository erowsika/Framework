 93     public function checkAccess() {
 94         $method = Base::instance()->router->getMethod();
 95         $auth = null;
 96         if (method_exists($this, 'access')) {
 97             $auth = Base::instance()->auth;
 98             $access = $this->access();
 99             foreach ($access as $rule) {
100                 $executes = $rule['executes'];
101                 
102                 $userRole = isset($rule['user']) ? $rule['user'] : $rule['role'];
103                 
104                 $userAuth = isset($rule['user']) ? $auth->getUser() : $auth->getRole();
105                 
106                 if (in_array($method, $executes)) {
107                     $accessType = reset($rule);
108                     $hasAuth = in_array($userAuth, $userRole);
109                     if ($accessType == 'grant' and $hasAuth) {
110                         return true;
111                     } else if ($accessType == 'revoke' and $hasAuth) {
112                         $this->outputJson(array("you don't have any such privileges to access this page"));
113                         Base::instance()->input->response($this->output, "application/json", 403);
114                     }
115                 }
116             }
117         }
118     }
