29         $template = str_replace('{controller}', $this->controller, $template);

31         $template = str_replace('{model_class}', $this->model_class, $template);

38     public function run() {

40         $action = Base::instance()->input->get('action');
41         switch ($action) {
42             case 'write_file':
43                 $filename = CONTROLLER_PATH . $this->controller . EXT_FILE;
44                 $this->write($filename, $code);
45                 break;
46             case 'download' :
47                 $code = $this->make();
48                 $filename = CONTROLLER_PATH . $this->controller . EXT_FILE;
49                 $this->download($filename, $code);
50                 exit();
51             default:
52                 break;
53         }
54         $db = Base::instance()->db->getConnection();
