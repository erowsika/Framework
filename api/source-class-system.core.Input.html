241     /**
242      * purify input
243      * @param string $data
244      * @return string
245      */
246     public function cleanInputs($data) {
247         $clean_input = array();
248         if (is_array($data)) {
249             foreach ($data as $k => $v) {
250                 $clean_input[$k] = $this->cleanInputs($v);
251             }
252         } else {
253             if (get_magic_quotes_gpc()) {
254                 $data = trim(stripslashes($data));
255             }
256             $data = strip_tags($data);
257             $clean_input = trim($data);
258         }
259         return $clean_input;
260     }
